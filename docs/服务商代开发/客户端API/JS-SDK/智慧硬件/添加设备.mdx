### 示例代码
```javascript
wx.invoke("addDevice", {
        "type": "qrcode",        // 设备发现方式，支持 qrcode/bluetooth/input
        "qrcode_url": "https://open.work.weixin.qq.com/connect?xxx",
        "sn": "TEST001"
    },function(res){
        if (res.err_msg == "addDevice:ok")
        {
            // 操作完成返回
        }
    }
);
```
