## 旧版jweixin调用
### 示例代码
```javascript
wx.onMenuShareTimeline({
    title: '', // 分享标题
    link: '', // 分享链接；在微信上分享时，该链接的域名必须与企业某个应用的可信域名一致
    imgUrl: '', // 分享图标
    enableIdTrans: 1, // 是否开启id转译，不填默认为0
    success: function () {
        // 用户确认分享后执行的回调函数
    },
    error: function(res) {
        if(res.errMsg.indexOf('no permission') > 0){
            alert('未agentConfig')
        }
    },
    cancel: function () {
        // 用户取消分享后执行的回调函数
    }
});
```
