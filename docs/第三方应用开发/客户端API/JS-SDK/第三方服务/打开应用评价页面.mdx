## 旧版jweixin调用
### 示例代码

```js
wx.invoke('openAppComment', {}, function(res){
 if(res.err_msg == "openAppComment:ok") {
  // 调用成功              
  }
  if(res.err_msg == "openAppComment:fail:no permission") {
  // 调用人身份不符合                 
  }
  if(res.err_msg == "openAppComment:fail:unknown app") {
  // 应用信息获取失败
  }
  if(res.err_msg == "openAppComment:fail:unsupported app type") {
  // 应用类型不符合要求
  }
  if(res.err_msg == "openAppComment:fail") {
  // 其它错误                  
  }      
})
```
